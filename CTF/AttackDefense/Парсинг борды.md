Для контроля статусов сервисов можно парсить борду организаторов.  
Скрипт на Python для парсинга кастомной A/D CTF борды на Plotly Dash и отправки уведомлений в Telegram-бота:  
```Python
#!/usr/bin/env python3
# parser.py

import html
import json
import re
import time
from datetime import datetime
from pathlib import Path
from typing import Any

import requests


# =========================
# CONFIG
# =========================

BOT_TOKEN = "" # Токен бота из @BotFather
CHAT_ID = "" # ChatId можно получить из https://api.telegram.org/bot<TOKEN>/getUpdates

TEAM_NAME = "" # Название команды на борде

BASE_URL = "https://url.com"
POLL_INTERVAL_SECONDS = 20

# Что слать в Telegram:
# "bad"    — только ухудшения: UP -> MUMBLE/DOWN/CORRUPT/CHECK_FAILED
# "status" — любые изменения статуса сервиса, включая восстановление
# "all"    — еще score, position, SLA, flags
NOTIFY_MODE = "status"

SEND_INITIAL_STATE = True

LATEST_JSON = "cyberctf_latest.json"
EVENTS_JSONL = "cyberctf_events.jsonl"


STATUS_BY_COLOR = {
    "#0a410a": "UP",
    "#17204d": "CORRUPT",
    "#9d4615": "MUMBLE",
    "#5d0e0e": "DOWN",
    "#d49f15": "CHECK_FAILED",
}

BAD_STATUSES = {"MUMBLE", "DOWN", "CORRUPT", "CHECK_FAILED", "UNKNOWN"}


# =========================
# DASH FETCH
# =========================

def fetch_dash_scoreboard(session: requests.Session, n_intervals: int) -> dict:
    payload = {
        "output": "body-div.children",
        "outputs": {
            "id": "body-div",
            "property": "children",
        },
        "inputs": [
            {
                "id": "update-interval",
                "property": "n_intervals",
                "value": n_intervals,
            }
        ],
        "state": [],
        "changedPropIds": [
            "update-interval.n_intervals",
        ],
    }

    response = session.post(
        f"{BASE_URL}/_dash-update-component",
        json=payload,
        timeout=10,
    )
    response.raise_for_status()
    return response.json()


# =========================
# DASH TREE HELPERS
# =========================

def props(node: Any) -> dict:
    if isinstance(node, dict):
        return node.get("props", {}) or {}
    return {}


def children(node: Any) -> Any:
    return props(node).get("children")


def node_type(node: Any) -> str | None:
    if isinstance(node, dict):
        return node.get("type")
    return None


def flat_text(obj: Any) -> list[str]:
    if obj is None:
        return []

    if isinstance(obj, (str, int, float)):
        return [str(obj)]

    if isinstance(obj, dict):
        return flat_text(children(obj))

    if isinstance(obj, list):
        out = []
        for item in obj:
            out.extend(flat_text(item))
        return out

    return [str(obj)]


def walk(obj: Any):
    yield obj

    if isinstance(obj, dict):
        for value in obj.values():
            yield from walk(value)
    elif isinstance(obj, list):
        for item in obj:
            yield from walk(item)


# =========================
# SCOREBOARD PARSER
# =========================

def get_root_children(data: dict) -> list[Any]:
    return data["response"]["body-div"]["children"]


def extract_round(root_children: list[Any]) -> int | None:
    text = flat_text(root_children)

    for i, item in enumerate(text):
        if "РАУНД" in item.upper() and i + 1 < len(text):
            try:
                return int(float(text[i + 1]))
            except ValueError:
                return None

    return None


def find_scoreboard_table(root_children: list[Any]) -> Any:
    for node in walk(root_children):
        if node_type(node) != "Table":
            continue

        text = " ".join(flat_text(node))

        if (
            "ПОЗИЦИЯ" in text
            and "КОМАНДА" in text
            and "ОБЩИЙ БАЛЛ" in text
        ):
            return node

    raise RuntimeError("scoreboard table not found")


def normalize_service_name(raw: str) -> str:
    # "[3] ServiceCTF" -> "ServiceCTF"
    return re.sub(r"^\[\d+\]\s*", "", raw).strip()


def parse_team_cell(cell: Any) -> tuple[str | None, str | None]:
    texts = [x.strip() for x in flat_text(cell) if x.strip()]

    name = None
    ip = None

    for text in texts:
        if re.fullmatch(r"\d+\.\d+\.\d+\.\d+", text):
            ip = text
        elif name is None:
            name = text

    return name, ip


def parse_service_cell(cell: Any) -> dict:
    style = props(cell).get("style") or {}
    color = str(style.get("backgroundColor", "")).lower()

    result = {
        "status": STATUS_BY_COLOR.get(color, "UNKNOWN"),
        "color": color,
        "sla": None,
        "fp": None,
        "flags_plus": None,
        "flags_minus": None,
    }

    for text in flat_text(cell):
        text = text.strip()

        match = re.search(r"SLA:\s*([\d.]+)%", text)
        if match:
            result["sla"] = float(match.group(1))
            continue

        match = re.search(r"FP:\s*([+-]?\d+(?:\.\d+)?)", text)
        if match:
            result["fp"] = float(match.group(1))
            continue

        match = re.search(r"⚑:\s*\+?(-?\d+)\s*/\s*([+-]?\d+)", text)
        if match:
            result["flags_plus"] = int(match.group(1))
            result["flags_minus"] = int(match.group(2))
            continue

    return result


def get_direct_children_by_type(node: Any, type_name: str) -> list[Any]:
    ch = children(node)
    if not isinstance(ch, list):
        ch = [ch]

    return [item for item in ch if node_type(item) == type_name]


def parse_scoreboard(data: dict) -> dict:
    root_children = get_root_children(data)
    round_number = extract_round(root_children)

    table = find_scoreboard_table(root_children)

    thead_list = get_direct_children_by_type(table, "Thead")
    tbody_list = get_direct_children_by_type(table, "Tbody")

    if not thead_list or not tbody_list:
        raise RuntimeError("unexpected scoreboard table structure")

    thead = thead_list[0]
    tbody = tbody_list[0]

    header_rows = get_direct_children_by_type(thead, "Tr")
    if not header_rows:
        raise RuntimeError("scoreboard header row not found")

    header_cells = children(header_rows[0])
    if not isinstance(header_cells, list):
        raise RuntimeError("scoreboard header cells not found")

    headers = [" ".join(flat_text(cell)).strip() for cell in header_cells]
    service_names = [normalize_service_name(h) for h in headers[5:]]

    rows = get_direct_children_by_type(tbody, "Tr")

    teams = []

    for row in rows:
        cells = children(row)

        if not isinstance(cells, list) or len(cells) < 5:
            continue

        position_texts = flat_text(cells[0])
        score_texts = flat_text(cells[4])

        if not position_texts or not score_texts:
            continue

        name, ip = parse_team_cell(cells[2])

        try:
            position = int(float(position_texts[0]))
            score = float(score_texts[0])
        except ValueError:
            continue

        team = {
            "position": position,
            "name": name,
            "ip": ip,
            "score": score,
            "services": {},
        }

        for service_name, service_cell in zip(service_names, cells[5:]):
            team["services"][service_name] = parse_service_cell(service_cell)

        teams.append(team)

    return {
        "ok": True,
        "round": round_number,
        "updated_at": int(time.time()),
        "services": service_names,
        "teams": teams,
    }


# =========================
# DIFF / EVENTS
# =========================

def find_team(parsed: dict, team_name: str) -> dict | None:
    for team in parsed["teams"]:
        if team["name"] and team["name"].lower() == team_name.lower():
            return team
    return None


def team_state(team: dict) -> dict:
    return {
        "position": team["position"],
        "score": team["score"],
        "ip": team["ip"],
        "services": {
            name: {
                "status": svc["status"],
                "sla": svc["sla"],
                "fp": svc["fp"],
                "flags_plus": svc["flags_plus"],
                "flags_minus": svc["flags_minus"],
            }
            for name, svc in team["services"].items()
        },
    }


def is_bad(status: str) -> bool:
    return status in BAD_STATUSES


def state_has_bad_services(state: dict) -> bool:
    return any(is_bad(svc["status"]) for svc in state["services"].values())


def diff_state(old: dict | None, new: dict) -> list[dict]:
    if old is None:
        return [{
            "kind": "initial",
            "level": "WARN" if state_has_bad_services(new) else "INFO",
            "message": "initial state loaded",
        }]

    events = []

    if old["position"] != new["position"]:
        events.append({
            "kind": "position",
            "level": "INFO",
            "message": f"position: {old['position']} → {new['position']}",
        })

    if old["score"] != new["score"]:
        events.append({
            "kind": "score",
            "level": "INFO",
            "message": f"score: {old['score']} → {new['score']}",
        })

    for service_name, new_svc in new["services"].items():
        old_svc = old["services"].get(service_name)

        if old_svc is None:
            events.append({
                "kind": "status",
                "level": "WARN" if is_bad(new_svc["status"]) else "INFO",
                "message": f"{service_name}: new service status {new_svc['status']}",
            })
            continue

        old_status = old_svc["status"]
        new_status = new_svc["status"]

        if old_status != new_status:
            if old_status != "UP" and new_status == "UP":
                level = "RECOVERY"
            elif is_bad(new_status):
                level = "WARN"
            else:
                level = "INFO"

            events.append({
                "kind": "status",
                "level": level,
                "message": f"{service_name}: {old_status} → {new_status}",
            })

        if old_svc["sla"] != new_svc["sla"]:
            events.append({
                "kind": "sla",
                "level": "INFO",
                "message": f"{service_name} SLA: {old_svc['sla']} → {new_svc['sla']}",
            })

        if (
            old_svc["flags_plus"] != new_svc["flags_plus"]
            or old_svc["flags_minus"] != new_svc["flags_minus"]
        ):
            events.append({
                "kind": "flags",
                "level": "INFO",
                "message": (
                    f"{service_name} flags: "
                    f"+{old_svc['flags_plus']}/-{old_svc['flags_minus']} → "
                    f"+{new_svc['flags_plus']}/-{new_svc['flags_minus']}"
                ),
            })

    return events


def should_send(events: list[dict]) -> bool:
    if not events:
        return False

    if not SEND_INITIAL_STATE and all(e["kind"] == "initial" for e in events):
        return False

    if NOTIFY_MODE == "bad":
        return any(e["level"] == "WARN" for e in events)

    if NOTIFY_MODE == "status":
        return any(e["kind"] in {"initial", "status"} for e in events)

    if NOTIFY_MODE == "all":
        return True

    return True


# =========================
# TELEGRAM
# =========================

def format_services(state: dict) -> str:
    lines = []

    for service_name, svc in state["services"].items():
        status = svc["status"]
        emoji = "✅" if status == "UP" else "🚨"

        lines.append(
            f"{emoji} <b>{html.escape(service_name)}</b>: "
            f"{html.escape(status)} "
            f"(SLA {svc['sla']}%, FP {svc['fp']}, "
            f"⚑ +{svc['flags_plus']} / -{svc['flags_minus']})"
        )

    return "\n".join(lines)


def format_telegram_message(team_name: str, round_number: int | None, state: dict, events: list[dict]) -> str:
    if any(e["level"] == "WARN" for e in events):
        title = "🚨 CyberCTF alert"
    elif any(e["level"] == "RECOVERY" for e in events):
        title = "✅ CyberCTF recovery"
    else:
        title = "ℹ️ CyberCTF update"

    event_lines = "\n".join(
        f"• {html.escape(e['message'])}"
        for e in events
    )

    return f"""<b>{title}</b>

<b>Team:</b> {html.escape(team_name)}
<b>Round:</b> {round_number}
<b>IP:</b> {html.escape(str(state["ip"]))}
<b>Position:</b> {state["position"]}
<b>Score:</b> {state["score"]}

<b>Changes:</b>
{event_lines}

<b>Services:</b>
{format_services(state)}
"""


def send_telegram(text: str) -> None:
    if BOT_TOKEN == "":
        raise RuntimeError("Set BOT_TOKEN in script")

    if CHAT_ID == "":
        raise RuntimeError("Set CHAT_ID in script")

    response = requests.post(
        f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage",
        json={
            "chat_id": CHAT_ID,
            "text": text[:3900],
            "parse_mode": "HTML",
            "disable_web_page_preview": True,
        },
        timeout=10,
    )

    response.raise_for_status()


# =========================
# LOGGING
# =========================

def now() -> str:
    return datetime.now().strftime("%H:%M:%S")


def append_jsonl(path: str, item: dict) -> None:
    with open(path, "a", encoding="utf-8") as f:
        f.write(json.dumps(item, ensure_ascii=False) + "\n")


def write_latest(parsed: dict, team: dict) -> None:
    latest = {
        "ok": True,
        "updated_at": int(time.time()),
        "round": parsed["round"],
        "team": team,
    }

    Path(LATEST_JSON).write_text(
        json.dumps(latest, indent=2, ensure_ascii=False),
        encoding="utf-8",
    )


# =========================
# MAIN LOOP
# =========================

def main():
    session = requests.Session()
    session.headers.update({
        "User-Agent": "Mozilla/5.0 CyberCTF-scoreboard-watcher/1.0",
        "Accept": "application/json,text/plain,*/*",
    })

    # Прогреваем сессию/cookies.
    session.get(BASE_URL, timeout=10).raise_for_status()

    last_state = None
    n_intervals = 1

    print(
        f"[+] watching team={TEAM_NAME}, interval={POLL_INTERVAL_SECONDS}s, notify={NOTIFY_MODE}",
        flush=True,
    )

    while True:
        try:
            raw = fetch_dash_scoreboard(session, n_intervals)
            parsed = parse_scoreboard(raw)

            team = find_team(parsed, TEAM_NAME)

            if team is None:
                message = f"team not found on scoreboard: {TEAM_NAME}"
                print(f"[{now()}] WARN {message}", flush=True)

                send_telegram(
                    f"🚨 <b>CyberCTF watcher warning</b>\n\n"
                    f"{html.escape(message)}"
                )

                time.sleep(POLL_INTERVAL_SECONDS)
                n_intervals += 1
                continue

            write_latest(parsed, team)

            current_state = team_state(team)
            events = diff_state(last_state, current_state)

            if events:
                print(
                    f"[{now()}] round={parsed['round']} "
                    f"team={TEAM_NAME} pos={current_state['position']} "
                    f"score={current_state['score']}",
                    flush=True,
                )

                for event in events:
                    log_item = {
                        "ts": int(time.time()),
                        "round": parsed["round"],
                        "team": TEAM_NAME,
                        **event,
                    }

                    append_jsonl(EVENTS_JSONL, log_item)
                    print(f"  [{event['level']}] {event['message']}", flush=True)

                if should_send(events):
                    send_telegram(
                        format_telegram_message(
                            TEAM_NAME,
                            parsed["round"],
                            current_state,
                            events,
                        )
                    )

            last_state = current_state

        except Exception as e:
            print(f"[{now()}] ERROR {e}", flush=True)

            try:
                send_telegram(
                    f"🚨 <b>CyberCTF watcher error</b>\n\n"
                    f"<code>{html.escape(str(e))}</code>"
                )
            except Exception as tg_error:
                print(f"[{now()}] TELEGRAM ERROR {tg_error}", flush=True)

        time.sleep(POLL_INTERVAL_SECONDS)
        n_intervals += 1


if __name__ == "__main__":
    main()
```
